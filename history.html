let cachedVersionOf originalFunction =
    let mutable cache = Map.empty
    fun key ->
        match cache |> Map.tryFind key  with
        | Some value -> value
        | None ->
            let value = originalFunction(key)
            cache <- cache |> Map.add key value 
            value
